C#. Fluent Assertions

Basic Assertions

Equal Assertions

result.Should().Be(expected);

Not Equal

result.Should().NotBe(unexpected);

Greater Than / Less Than

result.Should().BeGreaterThan(value);
result.Should().BeLessThan(value);

Null Assertions

object.Should().BeNull();
object.Should().NotBeNull();

Collection Assertions

Contain Items

collection.Should().Contain(expectedItem);

Contain Exactly

collection.Should().ContainInOrder(expectedItems);

Be Empty / Have Count

collection.Should().BeEmpty();
collection.Should().HaveCount(expectedCount);

Exception Assertions

Throw Exception

Action action = () => methodThatShouldThrow();

action.Should().Throw<ExpectedExceptionType>();

With message: To do:...

Object Assertions

Be Equivalent To

actualObject.Should().BeEquivalentTo(expectedObject);

Have Property

actualObject.Should().HaveProperty("PropertyName").Which.Should().Be(expectedValue);
